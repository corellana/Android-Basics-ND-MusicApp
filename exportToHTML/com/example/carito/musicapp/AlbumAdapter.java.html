1    package com.example.carito.musicapp; 
2     
3    import android.app.Activity; 
4    import android.content.Intent; 
5    import android.view.LayoutInflater; 
6    import android.view.View; 
7    import android.view.ViewGroup; 
8    import android.widget.ArrayAdapter; 
9    import android.widget.ImageView; 
10   import android.widget.TextView; 
11   import java.util.ArrayList; 
12    
13   public class AlbumAdapter extends ArrayAdapter<Album> { 
14    
15           /** 
16            * CONSTRUCTOR 
17            * @param context The current context. Used to inflate the layout file. 
18            * @param albums  A List of albums objects to display in a list. 
19            *                El constructor inicializa el objeto (AlbumAdapter) por ti. 
20            */ 
21           public AlbumAdapter(Activity context, ArrayList<Album> albums) { 
22               super(context, 0, albums); 
23           } 
24    
25           /** 
26            * * METHOD - GETVIEW 
27            * Provides a view for an AdapterView (ListView, GridView, etc.) 
28            * 
29            * @param position    The position in the list of data that should be displayed in the 
30            *                    list item view. 
31            * @param convertView The recycled view to populate. 
32            * @param parent      The parent ViewGroup that is used for inflation. 
33            * @return The View for the position in the AdapterView. 
34            */ 
35           @Override 
36           public View getView(final int position, View convertView, ViewGroup parent) { 
37               // List view llama al adapter pidiendole cada una de las vistas que 
38               // le corresponden a cada item del arreglo de canciones. 
39    
40               // Check if the existing view is being reused, otherwise inflate the view. 
41               // Convierte el layout en una vista. 
42               View albumListItemView = convertView; 
43               if (albumListItemView == null) { 
44                   albumListItemView = LayoutInflater.from(getContext()).inflate( 
45                           R.layout.albums_list_item, parent, false); 
46               } 
47               // Get the {@link Album} object located at this position in the list. 
48               final Album currentAlbum = getItem(position); 
49    
50               // Find the TextView in the albums_list_item.xml layout with the ID album_name_text_view 
51               TextView albumNameTextView = (TextView) albumListItemView.findViewById(R.id.album_name_text_view); 
52               // Get the Album Name from the current Album object and 
53               // set this text on the Album Name TextView 
54               albumNameTextView.setText(currentAlbum.getAlbumName()); 
55    
56               // Find the ImageView in the albums_list_item.xml layout with the ID album_icon_text_view 
57               ImageView albumIconImageView = (ImageView) albumListItemView.findViewById(R.id.icon_album_image_view); 
58               // Get the Artist Name from the current Song object and 
59               // set this text on the Artist Name TextView 
60               albumIconImageView.setImageResource(currentAlbum.getAlbumIcon()); 
61               /// Hasta aquí, el ListView mostrará todas los albums porque le pidió ayuda al Adapter que 
62               //  se sabe todas las canciones. Y el listview usará GetView para poder mostrar la lista completa. 
63    
64               // Ya configuramos el contenido, ahora configuraremos las acciones. 
65               albumListItemView.setOnClickListener(new View.OnClickListener() { 
66                   @Override 
67                   public void onClick(View v) { 
68                       Intent albumIntent = new Intent(getContext(), AlbumActivity.class); 
69                       // A current Album le pedimos el Album Name y el Album Icon 
70                       // y se lo pasamos al intent como un extra. 
71                       albumIntent.putExtra("Album Name", currentAlbum.getAlbumName()); 
72                       albumIntent.putExtra("Album Icon", currentAlbum.getAlbumIcon()); 
73   //                    albumIntent.putExtra("Song List", currentAlbum.getAlbumSongList()); 
74                       getContext().startActivity(albumIntent); 
75                   } 
76               }); 
77    
78               // Return the whole list item layout (containing 2 TextViews) 
79               // so that it can be shown in the ListView 
80               // Retornamos la vista configurada (tiene su layout, su contenido y acciones). 
81               return albumListItemView; 
82           } 
83   }
